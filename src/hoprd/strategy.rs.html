43fn validate_execution_interval(interval: &Duration) -> std::result::Result<(), ValidationError> {
44    if interval < &Duration::from_secs(10) {
45        Err(ValidationError::new(
46            "strategy execution interval must be at least 10 seconds",
47        ))
48    } else {
49        Ok(())
50    }

53/// Lists all possible strategies with their respective configurations.
54///
55/// This is a pure serde config type — it is used for YAML deserialization and
56/// carries no runtime behaviour. The runtime combinator is [`hopr_strategy::strategy::MultiStrategy`],
57/// which accepts any `Box<dyn Strategy + Send>`.
58#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, StrumDisplay, VariantNames)]
59#[strum(serialize_all = "snake_case")]
60pub enum StrategyKind {
61    #[cfg(feature = "runtime-tokio")]
62    AutoRedeeming(hopr_strategy::auto_redeeming::AutoRedeemingStrategyConfig),
63    #[cfg(feature = "runtime-tokio")]
64    AutoFunding(hopr_strategy::auto_funding::AutoFundingStrategyConfig),
65    #[cfg(feature = "runtime-tokio")]
66    ClosureFinalizer(hopr_strategy::channel_finalizer::ClosureFinalizerStrategyConfig),
67    #[cfg(feature = "runtime-tokio")]
68    ChannelLifecycle(Box<hopr_strategy::channel_lifecycle::ChannelLifecycleConfig>),
69    Multi(MultiStrategyConfig),
70    Passive,
71}
72
73impl validator::Validate for StrategyKind {
74    fn validate(&self) -> std::result::Result<(), validator::ValidationErrors> {
75        match self {
76            #[cfg(feature = "runtime-tokio")]
77            Self::AutoRedeeming(cfg) => cfg.validate(),
78            #[cfg(feature = "runtime-tokio")]
79            Self::AutoFunding(cfg) => cfg.validate(),
80            #[cfg(feature = "runtime-tokio")]
81            Self::ClosureFinalizer(cfg) => cfg.validate(),
82            #[cfg(feature = "runtime-tokio")]
83            Self::ChannelLifecycle(cfg) => cfg.validate(),
84            Self::Multi(cfg) => cfg.validate(),
85            Self::Passive => Ok(()),
86        }
87    }
88}

90/// Configuration options for the `MultiStrategy` group.
91#[derive(Debug, Clone, PartialEq, SmartDefault, Validate, Serialize, Deserialize)]
92#[serde(deny_unknown_fields)]
93pub struct MultiStrategyConfig {
94    /// Indicate whether the `MultiStrategy` can contain another `MultiStrategy`.
95    ///
96    /// Default is `true`. Nesting is limited to one level: when this is `true`, nested
97    /// `Multi` groups have their own `allow_recursive` forced to `false`, so three-deep
98    /// nesting is silently flattened.
99    #[default = true]
100    #[serde(default = "just_true")]
101    pub allow_recursive: bool,
102
103    /// Execution interval for periodic scans within each sub-strategy.
104    ///
105    /// Default is 60 seconds, minimum is 10 seconds.
106    #[default(sixty_seconds())]
107    #[serde(default = "sixty_seconds", with = "humantime_serde")]
108    #[validate(custom(function = "validate_execution_interval"))]
109    pub execution_interval: Duration,
110
111    /// Configuration of individual sub-strategies.
112    ///
113    /// Default is empty, which makes the `MultiStrategy` behave as passive.
114    #[default(_code = "vec![]")]
115    #[serde(default = "empty_strategies")]
116    #[validate(nested)]
117    pub strategies: Vec<StrategyKind>,
118}

120/// Default HOPRd strategy configuration.
121///
122/// ## Strategies included
123/// - `AutoRedeeming` *(requires `runtime-tokio` feature)*: redeems single tickets on channel close if worth at least 1
124///   wxHOPR.
125/// - `ChannelLifecycle` *(requires `runtime-tokio` feature)*: unified strategy that automatically opens, funds,
126///   tops up, closes, and finalizes outgoing payment channels based on peer connectivity and quality.
127///
128/// When `runtime-tokio` is not enabled, returns an empty `MultiStrategyConfig` (passive behaviour).
129pub fn hopr_default_strategies() -> MultiStrategyConfig {
130    #[cfg(feature = "runtime-tokio")]
131    {
132        use hopr_strategy::auto_redeeming::AutoRedeemingStrategyConfig;
133        return MultiStrategyConfig {
134            allow_recursive: false,
135            execution_interval: Duration::from_secs(60),
136            strategies: vec![
137                StrategyKind::AutoRedeeming(AutoRedeemingStrategyConfig {
138                    redeem_on_winning: true,
139                    ..Default::default()
140                }),
141                StrategyKind::ChannelLifecycle(Box::default()),
142            ],
143        };
144    }
145    #[allow(unreachable_code)]
146    MultiStrategyConfig::default()
147}

149/// Builds a [`MultiStrategy`] from a [`MultiStrategyConfig`] and a node reference.
150///
151/// Maps each [`StrategyKind`] variant to its concrete builder, wires in the node,
152/// and returns a single `Box<dyn Strategy + Send>` that runs all sub-strategies
153/// concurrently.
154///
155/// External strategies can be composed by building this result first, then wrapping
156/// it with additional strategies in a new `MultiStrategy::new(...)` call at the
157/// call site.
158pub fn build_strategies<N>(cfg: &MultiStrategyConfig, node: Arc<N>) -> Box<dyn Strategy + Send>
159where
160    N: ActionableEventSource
161        + HasChainApi<
162            ChainApi: ChainReadAccountOperations
163                          + ChainReadChannelOperations
164                          + ChainReadSafeOperations
165                          + ChainValues
166                          + ChainWriteChannelOperations
167                          + ChainWriteTicketOperations
168                          + Clone
169                          + Send
170                          + Sync
171                          + 'static,
172        > + HasGraphView
173        + HasNetworkView
174        + HasTicketManagement<TicketManager: TicketManagement + Clone + Send + Sync + 'static>
175        + Send
176        + Sync
177        + 'static,
178{
179    // Seed all gauges to 0 exactly once at the top level — recursive calls via
180    // StrategyKind::Multi must not reset them or they would clobber values set
181    // by earlier iterations of the outer loop.
182    #[cfg(all(feature = "telemetry", not(test)))]
183    StrategyKind::VARIANTS
184        .iter()
185        .for_each(|s| METRIC_ENABLED_STRATEGIES.set(&[*s], 0_f64));
186
187    build_strategies_inner(cfg, node)
188}

190fn build_strategies_inner<N>(cfg: &MultiStrategyConfig, node: Arc<N>) -> Box<dyn Strategy + Send>
191where
192    N: ActionableEventSource
193        + HasChainApi<
194            ChainApi: ChainReadAccountOperations
195                          + ChainReadChannelOperations
196                          + ChainReadSafeOperations
197                          + ChainValues
198                          + ChainWriteChannelOperations
199                          + ChainWriteTicketOperations
200                          + Clone
201                          + Send
202                          + Sync
203                          + 'static,
204        > + HasGraphView
205        + HasNetworkView
206        + HasTicketManagement<TicketManager: TicketManagement + Clone + Send + Sync + 'static>
207        + Send
208        + Sync
209        + 'static,
210{
211    let mut strategies = Vec::<Box<dyn Strategy + Send>>::new();
212
213    for strategy in cfg.strategies.iter() {
214        match strategy {
215            #[cfg(feature = "runtime-tokio")]
216            StrategyKind::AutoRedeeming(sub_cfg) => strategies.push(
217                hopr_strategy::auto_redeeming::AutoRedeemingStrategy::new(
218                    *sub_cfg,
219                    cfg.execution_interval,
220                )
221                .build(Arc::clone(&node)),
222            ),
223            #[cfg(feature = "runtime-tokio")]
224            StrategyKind::AutoFunding(sub_cfg) => strategies.push(
225                hopr_strategy::auto_funding::AutoFundingStrategy::new(
226                    *sub_cfg,
227                    cfg.execution_interval,
228                )
229                .build(Arc::clone(&node)),
230            ),
231            #[cfg(feature = "runtime-tokio")]
232            StrategyKind::ClosureFinalizer(sub_cfg) => strategies.push(
233                hopr_strategy::channel_finalizer::ClosureFinalizerStrategy::new(
234                    *sub_cfg,
235                    cfg.execution_interval,
236                )
237                .build(Arc::clone(&node)),
238            ),
239            // ChannelLifecycle owns its own tick cadence via
240            // ChannelLifecycleConfig::tick_interval and runs as an independent
241            // async task; cfg.execution_interval does not apply to it.
242            #[cfg(feature = "runtime-tokio")]
243            StrategyKind::ChannelLifecycle(sub_cfg) => strategies.push(
244                hopr_strategy::channel_lifecycle::ChannelLifecycleStrategy::new(
245                    (**sub_cfg).clone(),
246                )
247                .build(Arc::clone(&node)),
248            ),
249            StrategyKind::Multi(sub_cfg) => {
250                if cfg.allow_recursive {
251                    let mut sub = sub_cfg.clone();
252                    sub.allow_recursive = false;
253                    strategies.push(build_strategies_inner(&sub, Arc::clone(&node)));
254                } else {
255                    tracing::error!("recursive multi-strategy not allowed and skipped");
256                    continue; // skip the telemetry update: nothing was actually built
257                }
258            }
259            StrategyKind::Passive => {} // passive = empty sub-strategy list
260        }
261
262        #[cfg(all(feature = "telemetry", not(test)))]
263        if !matches!(strategy, StrategyKind::Passive) {
264            METRIC_ENABLED_STRATEGIES.set(&[&strategy.to_string()], 1_f64);
265        }
266    }
267
268    Box::new(MultiStrategy::new(strategies))
