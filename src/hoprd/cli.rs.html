18/// Sentinel host for the session listen host that triggers local-IP auto-detection.
19///
20/// Used by container deployments that cannot know their routable IP ahead of time:
21/// `auto` (or `auto:<port>`) is resolved to the node's primary non-loopback IPv4 in
22/// [`HoprdConfig::try_from`]. Native deployments that leave the value unset keep the
23/// `127.0.0.1:0` default.
24pub const SESSION_LISTEN_HOST_AUTO: &str = "auto";

26fn parse_host(s: &str) -> Result<HostConfig, String> {
27    let host = s.split_once(':').map_or(s, |(h, _)| h);
28    if !(validator::ValidateIp::validate_ipv4(&host) || looks_like_domain(host)) {
29        return Err(format!(
30            "Given string {s} is not a valid host, should have a format: <ip>:<port> or <domain>(:<port>)"
31        ));
32    }
33
34    HostConfig::from_str(s)
35}

37/// Parses the session listen host, additionally accepting the [`SESSION_LISTEN_HOST_AUTO`]
38/// sentinel (`auto` or `auto:<port>`). The sentinel is carried as a `Domain` host and
39/// resolved to a concrete IP when the effective config is built.
40fn parse_session_listen_host(s: &str) -> Result<HostConfig, String> {
41    let (host, port) = match s.split_once(':') {
42        Some((h, p)) => (h, Some(p)),
43        None => (s, None),
44    };
45
46    if host.eq_ignore_ascii_case(SESSION_LISTEN_HOST_AUTO) {
47        let port = match port {
48            Some(p) => p
49                .parse::<u16>()
50                .map_err(|e| format!("invalid port '{p}' in session listen host: {e}"))?,
51            None => 0,
52        };
53        return Ok(HostConfig {
54            address: HostType::Domain(SESSION_LISTEN_HOST_AUTO.to_string()),
55            port,
56        });
57    }
58
59    parse_host(s)
60}

62/// Detects the node's primary non-loopback IPv4 address.
63///
64/// Mirrors the container entrypoint's historical `hostname -i` behavior: it returns the
65/// first non-loopback IPv4 reported by the host's network interfaces. Used to resolve the
66/// [`SESSION_LISTEN_HOST_AUTO`] sentinel into a concrete, advertisable address.
67fn detect_local_ipv4() -> anyhow::Result<IpAddr> {
68    if_addrs::get_if_addrs()
69        .context("failed to enumerate network interfaces")?
70        .into_iter()
71        .map(|iface| iface.ip())
72        .find(|ip| ip.is_ipv4() && !ip.is_loopback())
73        .ok_or_else(|| anyhow!("no non-loopback IPv4 address found for session listen host"))
74}
75
76fn resolve_session_listen_host(
77    host: HostConfig,
78    detect_local_ipv4: impl FnOnce() -> anyhow::Result<IpAddr>,
79) -> anyhow::Result<std::net::SocketAddr> {
80    match host.address {
81        HostType::IPv4(addr) => IpAddr::from_str(&addr)
82            .map(|ip| std::net::SocketAddr::new(ip, host.port))
83            .map_err(|_| anyhow!("invalid default session listen IP address")),
84        HostType::Domain(ref d) if d.eq_ignore_ascii_case(SESSION_LISTEN_HOST_AUTO) => {
85            detect_local_ipv4().map(|ip| std::net::SocketAddr::new(ip, host.port))
86        }
87        HostType::Domain(_) => Err(anyhow!("default session listen must be an IP")),
88    }
89}

91fn parse_api_token(mut s: &str) -> Result<String, String> {
92    if s.len() < MINIMAL_API_TOKEN_LENGTH {
93        return Err(format!(
94            "Length of API token is too short, minimally required {MINIMAL_API_TOKEN_LENGTH} but given {}",
95            s.len()
96        ));
97    }
98
99    match (s.starts_with('\''), s.ends_with('\'')) {
100        (true, true) => {
101            s = s
102                .strip_prefix('\'')
103                .ok_or("failed to parse strip prefix part")?;
104            s = s
105                .strip_suffix('\'')
106                .ok_or("failed to parse strip suffix part")?;
107
108            Ok(s.into())
109        }
110        (true, false) => Err("Found leading quote but no trailing quote".into()),
111        (false, true) => Err("Found trailing quote but no leading quote".into()),
112        (false, false) => Ok(s.into()),
113    }
114}

312impl TryFrom<CliArgs> for HoprdConfig {
313    type Error = anyhow::Error;
314
315    fn try_from(value: CliArgs) -> Result<Self, Self::Error> {
316        let mut cfg: HoprdConfig = if let Some(cfg_path) = value.configuration_file_path {
317            tracing::debug!(cfg_path, "fetching configuration from file");
318            let yaml_configuration = std::fs::read_to_string(cfg_path.as_str())
319                .with_context(|| format!("failed to read config file '{cfg_path}'"))?;
320            serde_saphyr::from_str(&yaml_configuration).context("failed to parse config YAML")?
321        } else {
322            tracing::debug!("loading default configuration");
323            HoprdConfig::default()
324        };
325
326        // host
327        if let Some(x) = value.host {
328            cfg.hopr.host = x
329        };
330
331        // hopr.transport
332        if value.test_announce_local_addresses > 0 {
333            cfg.hopr.network.announce_local_addresses = true;
334        }
335        if value.test_prefer_local_addresses > 0 {
336            cfg.hopr.network.prefer_local_addresses = true;
337        }
338
339        if let Some(host) = value.default_session_listen_host {
340            cfg.session_ip_forwarding.default_entry_listen_host =
341                resolve_session_listen_host(host, detect_local_ipv4)?;
342        }
343        if value.enable_explicit_path_sessions {
344            cfg.api.enable_explicit_path_sessions = true;
345        }
346
347        // db
348        if let Some(data) = value.data {
349            cfg.db.data = data
350        }
351        if value.init > 0 {
352            cfg.db.initialize = true;
353        }
354        if value.force_init > 0 {
355            cfg.db.force_initialize = true;
356        }
357
358        // api
359        if value.api > 0 {
360            cfg.api.enable = true;
361        }
362        if value.disable_api_authentication > 0 && cfg.api.auth != Auth::None {
363            cfg.api.auth = Auth::None;
364        };
365        if let Some(x) = value.api_token {
366            cfg.api.auth = Auth::Token(x);
367        };
368        if let Some(x) = value.api_host {
369            cfg.api.host = HostConfig::from_str(
370                format!("{}:{}", x.as_str(), hoprd_api::config::DEFAULT_API_PORT).as_str(),
371            )
372            .map_err(|e| anyhow!("invalid api host: {e}"))?;
373        }
374        if let Some(x) = value.api_port {
375            cfg.api.host.port = x
376        };
377
378        // probe
379        if let Some(x) = value.probe_recheck_threshold {
380            cfg.hopr.network.probe_recheck_threshold = std::time::Duration::from_secs(x)
381        };
382
383        // identity
384        if let Some(identity) = value.identity {
385            cfg.identity.file = identity;
386        }
387        if let Some(x) = value.password {
388            cfg.identity.password = x
389        };
390        if let Some(x) = value.private_key {
391            cfg.identity.private_key = Some(x)
392        };
393
394        // chain
395        if value.announce > 0 {
396            cfg.hopr.announce = true;
397        }
398
399        if let Some(x) = value.blokli_url {
400            cfg.blokli_url = x;
401        }
402
403        if let Some(x) = value.safe_address {
404            cfg.hopr.safe_module.safe_address =
405                Address::from_str(&x).with_context(|| format!("invalid safe address '{x}'"))?
406        };
407        if let Some(x) = value.module_address {
408            cfg.hopr.safe_module.module_address =
409                Address::from_str(&x).with_context(|| format!("invalid module address '{x}'"))?
410        };
411
412        // additional updates
413        let home_symbol = '~';
414        if cfg.db.data.starts_with(home_symbol) {
415            cfg.db.data = home::home_dir()
416                .map(|h| h.as_path().display().to_string())
417                .expect("home dir for a user must be specified")
418                + &cfg.db.data[1..];
419        }
420        if cfg.identity.file.starts_with(home_symbol) {
421            cfg.identity.file = home::home_dir()
422                .map(|h| h.as_path().display().to_string())
423                .expect("home dir for a user must be specified")
424                + &cfg.identity.file[1..];
425        }
426
427        Ok(cfg)
428    }

440    #[test]
441    fn auto_sentinel_resolves_to_non_loopback_ipv4_with_given_port() -> anyhow::Result<()> {
442        let args = CliArgs::try_parse_from(["hoprd", "--defaultSessionListenHost", "auto:1234"])?;
443        let host = resolve_session_listen_host(
444            args.default_session_listen_host
445                .expect("session listen host must be parsed"),
446            || Ok("10.20.30.40".parse()?),
447        )?;
448
449        assert!(host.is_ipv4(), "expected an IPv4 address, got {host}");
450        assert!(
451            !host.ip().is_loopback(),
452            "expected a non-loopback address, got {host}"
453        );
454        assert_eq!(host.port(), 1234);
455
456        Ok(())
457    }
