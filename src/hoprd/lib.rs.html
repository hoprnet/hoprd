1//! HOPR node daemon binary. Runs the HOPR protocol (via [`hopr-lib`](https://github.com/hoprnet/hoprnet))
2//! and exposes a REST API for node management.
3//!
4//! When the REST API is enabled, interactive API docs are available at:
5//! - `http://localhost:3001/scalar` (Scalar UI)
6//! - `http://localhost:3001/swagger-ui` (Swagger UI)
7//!
8//! ## Usage
9//! See `hoprd --help` for the full list of options.

58pub async fn main_inner(cfg: HoprdConfig, hopr_keys: HoprKeys) -> anyhow::Result<()> {
59    use hopr_lib::api::types::primitive::traits::ToHex as _;
60
61    #[cfg(all(target_os = "linux", feature = "allocator-jemalloc-stats"))]
62    let _jemalloc_stats = jemalloc_stats::JemallocStats::start().await;
63
64    if cfg!(debug_assertions) {
65        tracing::warn!("Executable was built using the DEBUG profile.");
66    } else {
67        tracing::info!("Executable was built using the RELEASE profile.");
68    }
69
70    let git_hash = option_env!("VERGEN_GIT_SHA").unwrap_or("unknown");
71    tracing::info!(
72        version = env!("CARGO_PKG_VERSION"),
73        hash = git_hash,
74        cfg = cfg.as_redacted_string()?,
75        "Starting HOPR daemon"
76    );
77
78    if std::env::var("DAPPNODE")
79        .map(|v| v.to_lowercase() == "true")
80        .unwrap_or(false)
81    {
82        tracing::info!("The HOPRd node appears to run on DappNode");
83    }
84
85    let mut hopr_lib_cfg: HoprLibConfig = cfg.hopr.clone().into();
86    update_hopr_lib_config_from_env_vars(&mut hopr_lib_cfg)?;
87
88    tracing::info!(
89        packet_key = Keypair::public(&hopr_keys.packet_key).to_peerid_str(),
90        blockchain_address = Keypair::public(&hopr_keys.chain_key).to_address().to_hex(),
91        "Node public identifiers"
92    );
93
94    // Create the node instance
95    tracing::info!("creating the HOPRd node instance from hopr-lib");
96
97    let mut processes = AbortableList::<HoprdProcess>::default();
98
99    let mut chain_connector = create_trustful_hopr_blokli_connector(
100        &hopr_keys.chain_key,
101        BlockchainConnectorConfig {
102            connection_sync_timeout: std::time::Duration::from_mins(1),
103            sync_tolerance: 90,
104            tx_timeout_multiplier: std::env::var("HOPR_TX_TIMEOUT_MULTIPLIER")
105                .ok()
106                .and_then(|p| {
107                    p.parse()
108                        .inspect_err(|error| tracing::warn!(%error, "failed to parse HOPR_TX_TIMEOUT_MULTIPLIER"))
109                        .ok()
110                })
111                .unwrap_or_else(|| BlockchainConnectorConfig::default().tx_timeout_multiplier),
112        },
113        BlokliClient::new(
114            cfg.blokli_url.parse()?,
115            blokli_client::BlokliClientConfig {
116                timeout: std::time::Duration::from_secs(30),
117                stream_reconnect_timeout: std::time::Duration::from_secs(30),
118                subscription_stream_restart_delay: Some(std::time::Duration::from_secs(1)),
119                // Allow local clusters to skip the blokli version compatibility check:
120                // some dev images (e.g. bloklid-anvil) do index Safe events but don't
121                // yet advertise the IndexesSafeEvents feature flag in their API response.
122                auto_compatibility_check: std::env::var("HOPR_BLOKLI_NO_COMPAT_CHECK").is_err(),
123                ..Default::default()
124            },
125        ),
126        cfg.hopr.safe_module.module_address,
127    )
128    .await?;
129    chain_connector.connect().await?;
130    let chain_connector = Arc::new(chain_connector);
131
132    let prober_cfg = hopr_ct_full_network::ProberConfig {
133        interval: cfg.hopr.network.probe_interval,
134        shuffle_ttl: cfg.hopr.network.probe_interval * 2,
135        ..Default::default()
136    };
137    prober_cfg.validate_against_probe_timeout(hopr_lib_cfg.protocol.probe.timeout)?;
138
139    let (ticket_manager, ticket_factory) = ticket_manager_from_chain(&chain_connector)
140        .await
141        .map_err(|e| anyhow::anyhow!("failed to seed ticket manager: {e}"))?;
142
143    let path_cfg = hopr_lib_cfg.protocol.path_planner;
144    let graph: SharedChannelGraph = Arc::new(ChannelGraph::with_edge_params(
145        *hopr_keys.packet_key.public(),
146        path_cfg.edge_penalty,
147        path_cfg.min_ack_rate,
148    ));
149    let graph_for_ct = graph.clone();
150    let safe_address = hopr_lib_cfg.safe_module.safe_address;
151    let module_address = hopr_lib_cfg.safe_module.module_address;
152    #[cfg(feature = "session-server")]
153    let server = HoprServerIpForwardingReactor::new(
154        hopr_keys.packet_key.clone(),
155        cfg.session_ip_forwarding.clone(),
156    );
157
158    let builder = HoprBuilder
159        .with_identity(&hopr_keys.chain_key, &hopr_keys.packet_key)
160        .with_config(hopr_lib_cfg)
161        .with_safe_module(&safe_address, &module_address)
162        .with_chain_api(move |_ctx| chain_connector)
163        .with_graph(move |_ctx| graph)
164        .with_network(move |ctx| {
165            Box::pin(async move {
166                let peer_discovery_rx = ctx.take_peer_discovery_rx().ok_or(
167                    hopr_lib::errors::HoprLibError::BuilderError("peer_discovery_rx already taken"),
168                )?;
169                let multiaddresses = vec![
170                    (&ctx.cfg.host)
171                        .try_into()
172                        .map_err(hopr_lib::errors::HoprLibError::TransportError)?,
173                ];
174                let nb = HoprLibp2pNetworkBuilder::new(
175                    peer_discovery_rx
176                        .map(|(peer_id, addrs)| PeerDiscovery::Announce(peer_id, addrs)),
177                );
178                nb.build(
179                    &ctx.packet_key,
180                    multiaddresses,
181                    "/hopr/mix/1.1.0",
182                    ctx.cfg.protocol.transport.prefer_local_addresses,
183                )
184                .await
185                .map_err(|e| hopr_lib::errors::HoprLibError::GeneralError(e.to_string()))
186            })
187        })
188        .with_cover_traffic(move |ctx| {
189            hopr_ct_full_network::FullNetworkDiscovery::new(
190                *ctx.packet_key.public(),
191                prober_cfg,
192                graph_for_ct,
193            )
194        });
195
196    #[cfg(feature = "session-server")]
197    let node = Arc::new(
198        builder
199            .with_session_server(server)
200            .build_full(ticket_manager, ticket_factory)
201            .await?,
202    );
203    #[cfg(not(feature = "session-server"))]
204    let node = Arc::new(builder.build_full(ticket_manager, ticket_factory).await?);
205
206    if cfg.api.enable {
207        let list = init_rest_api(&cfg, node.clone()).await?;
208        processes.extend_from(list);
209    }
210
211    tracing::debug!("initializing strategies");
212    let mut multi_strategy = crate::strategy::build_strategies(&cfg.strategy, Arc::clone(&node));
213    tracing::debug!(strategy = %multi_strategy, "initialized strategies");
214
215    tracing::debug!("starting up strategies");
216    processes.insert(
217        HoprdProcess::Strategies,
218        tokio::spawn(async move {
219            if let Err(e) = multi_strategy.run().await {
220                tracing::error!(%e, "strategy terminated with error");
221            }
222        }),
223    );
224
225    let mut signals = Signals::new([Signal::Hup, Signal::Int, Signal::Term])
226        .context("failed to register signal handlers")?;
227    while let Some(Ok(signal)) = signals.next().await {
228        match signal {
229            Signal::Hup => {
230                tracing::info!("Received the HUP signal... not doing anything");
231            }
232            Signal::Int | Signal::Term => {
233                tracing::error!(signal = ?signal, "Received a termination signal... tearing down the node");
234                // Explicitly tear down running processes here
235                drop(node);
236                drop(processes);
237
238                tracing::error!(signal = ?signal, "All processes stopped... emulating the default signal handler...");
239                low_level::emulate_default_handler(signal as i32)?;
240                tracing::error!("Shutting down!");
241                break;
242            }
243            _ => {
244                tracing::error!(signal = ?signal, "Received an unhandled signal... emulating the default signal handler...");
245                low_level::emulate_default_handler(signal as i32)?;
246            }
247        }
248    }
249
250    Ok(())
251}

253async fn init_rest_api(
254    cfg: &HoprdConfig,
255    hopr: Arc<HoprNode>,
256) -> anyhow::Result<AbortableList<HoprdProcess>> {
257    let node_cfg_value =
258        serde_json::to_value(cfg.as_redacted()).context("failed to serialize redacted config")?;
259
260    let api_cfg = cfg.api.clone();
261
262    let listen_address = match &cfg.api.host.address {
263        hopr_lib::config::HostType::IPv4(a) | hopr_lib::config::HostType::Domain(a) => {
264            format!("{a}:{}", cfg.api.host.port)
265        }
266    };
267
268    let api_listener = tokio::net::TcpListener::bind(&listen_address)
269        .await
270        .map_err(|e| {
271            hopr_lib::errors::HoprLibError::GeneralError(format!(
272                "REST API bind failed for {listen_address}: {e}"
273            ))
274        })?;
275
276    tracing::info!(listen_address, "Running a REST API");
277
278    let session_listener_sockets = Arc::new(hopr_utils_session::ListenerJoinHandles::default());
279
280    let mut processes = AbortableList::<HoprdProcess>::default();
281
282    processes.insert(
283        HoprdProcess::ListenerSocket,
284        session_listener_sockets.clone(),
285    );
286
287    let cfg_clone = cfg.clone();
288    let (proc, abort_handle) = abortable(
289        async move {
290            if let Err(e) = serve_api(RestApiParameters {
291                listener: api_listener,
292                hoprd_cfg: node_cfg_value,
293                cfg: api_cfg,
294                hopr,
295                session_listener_sockets,
296                default_session_listen_host: cfg_clone
297                    .session_ip_forwarding
298                    .default_entry_listen_host,
299            })
300            .await
301            {
302                tracing::error!(error = %e, "the REST API server could not start")
303            }
304        }
305        .inspect(|_| {
306            tracing::warn!(
307                task = "hoprd - REST API",
308                "long-running background task finished"
309            )
310        }),
311    );
312    let _jh = tokio::spawn(proc);
313    processes.insert(HoprdProcess::RestApi, abort_handle);
314
315    Ok(processes)
316}
317fn update_hopr_lib_config_from_env_vars(cfg: &mut HoprLibConfig) -> anyhow::Result<()> {
318    cfg.protocol.packet.pipeline.output_concurrency = std::env::var("HOPR_INTERNAL_OUT_PACKET_PIPELINE_CONCURRENCY")
319        .ok()
320        .and_then(|p| {
321            p.parse()
322                .inspect_err(
323                    |error| tracing::warn!(%error, "failed to parse HOPR_INTERNAL_OUT_PACKET_PIPELINE_CONCURRENCY"),
324                )
325                .ok()
326        });
327
328    cfg.protocol.packet.pipeline.input_concurrency = std::env::var("HOPR_INTERNAL_IN_PACKET_PIPELINE_CONCURRENCY")
329        .ok()
330        .and_then(|p| {
331            p.parse()
332                .inspect_err(
333                    |error| tracing::warn!(%error, "failed to parse HOPR_INTERNAL_IN_PACKET_PIPELINE_CONCURRENCY"),
334                )
335                .ok()
336        });
337
338    #[cfg(feature = "session-server")]
339    if let Some(cap) = std::env::var("HOPR_INTERNAL_SESSION_INCOMING_CAPACITY")
340        .ok()
341        .and_then(|s| {
342            s.trim()
343                .parse::<usize>()
344                .inspect_err(
345                    |error| tracing::warn!(%error, "failed to parse HOPR_INTERNAL_SESSION_INCOMING_CAPACITY"),
346                )
347                .ok()
348        })
349        .filter(|&c| c > 0)
350    {
351        cfg.incoming_session_capacity = cap;
352    }
353
354    #[cfg(debug_assertions)]
355    if let Ok(v) = std::env::var("HOPR_TEST_DISABLE_CHECKS") {
356        cfg.disable_protocol_checks = v.to_lowercase() == "true";
357    }
358
359    Ok(())
360}

367    static ENV_LOCK: std::sync::Mutex<()> = std::sync::Mutex::new(());
368
369    const OUT_VAR: &str = "HOPR_INTERNAL_OUT_PACKET_PIPELINE_CONCURRENCY";
370    const IN_VAR: &str = "HOPR_INTERNAL_IN_PACKET_PIPELINE_CONCURRENCY";
371
372    struct EnvGuard(Vec<&'static str>);
373
374    impl Drop for EnvGuard {
375        fn drop(&mut self) {
376            for k in &self.0 {
377                // Safety: ENV_LOCK is held for the duration of each test, serializing all env access
378                unsafe { std::env::remove_var(k) };
379            }
380        }
381    }
382
383    fn setup_env(
384        vars: &[(&'static str, Option<&str>)],
385    ) -> (std::sync::MutexGuard<'static, ()>, EnvGuard) {
386        let guard = ENV_LOCK.lock().unwrap_or_else(|e| e.into_inner());
387        for (k, v) in vars {
388            match v {
389                // Safety: ENV_LOCK serializes all env access across tests in this module
390                Some(val) => unsafe { std::env::set_var(k, val) },
391                None => unsafe { std::env::remove_var(k) },
392            }
393        }
394        let keys = vars.iter().map(|(k, _)| *k).collect();
395        (guard, EnvGuard(keys))
396    }
