1//! HOPR node daemon binary. Runs the HOPR protocol (via [`hopr-lib`](https://github.com/hoprnet/hoprnet))
2//! and exposes a REST API for node management.
3//!
4//! When the REST API is enabled, interactive API docs are available at:
5//! - `http://localhost:3001/scalar` (Scalar UI)
6//! - `http://localhost:3001/swagger-ui` (Swagger UI)
7//!
8//! ## Usage
9//! See `hoprd --help` for the full list of options.

56pub async fn main_inner(cfg: HoprdConfig, hopr_keys: HoprKeys) -> anyhow::Result<()> {
57    use hopr_lib::api::types::primitive::traits::ToHex as _;
58
59    #[cfg(all(target_os = "linux", feature = "allocator-jemalloc-stats"))]
60    let _jemalloc_stats = jemalloc_stats::JemallocStats::start().await;
61
62    if cfg!(debug_assertions) {
63        tracing::warn!("Executable was built using the DEBUG profile.");
64    } else {
65        tracing::info!("Executable was built using the RELEASE profile.");
66    }
67
68    let git_hash = option_env!("VERGEN_GIT_SHA").unwrap_or("unknown");
69    tracing::info!(
70        version = hopr_lib::constants::APP_VERSION,
71        hash = git_hash,
72        cfg = cfg.as_redacted_string()?,
73        "Starting HOPR daemon"
74    );
75
76    if std::env::var("DAPPNODE")
77        .map(|v| v.to_lowercase() == "true")
78        .unwrap_or(false)
79    {
80        tracing::info!("The HOPRd node appears to run on DappNode");
81    }
82
83    let mut hopr_lib_cfg: HoprLibConfig = cfg.hopr.clone().into();
84    update_hopr_lib_config_from_env_vars(&mut hopr_lib_cfg)?;
85
86    tracing::info!(
87        packet_key = Keypair::public(&hopr_keys.packet_key).to_peerid_str(),
88        blockchain_address = Keypair::public(&hopr_keys.chain_key).to_address().to_hex(),
89        "Node public identifiers"
90    );
91
92    // Create the node instance
93    tracing::info!("creating the HOPRd node instance from hopr-lib");
94
95    let mut processes = AbortableList::<HoprdProcess>::default();
96
97    let mut chain_connector = create_trustful_hopr_blokli_connector(
98        &hopr_keys.chain_key,
99        BlockchainConnectorConfig {
100            connection_sync_timeout: std::time::Duration::from_mins(1),
101            sync_tolerance: 90,
102            tx_timeout_multiplier: std::env::var("HOPR_TX_TIMEOUT_MULTIPLIER")
103                .ok()
104                .and_then(|p| {
105                    p.parse()
106                        .inspect_err(|error| tracing::warn!(%error, "failed to parse HOPR_TX_TIMEOUT_MULTIPLIER"))
107                        .ok()
108                })
109                .unwrap_or_else(|| BlockchainConnectorConfig::default().tx_timeout_multiplier),
110        },
111        BlokliClient::new(
112            cfg.blokli_url.parse()?,
113            blokli_client::BlokliClientConfig {
114                timeout: std::time::Duration::from_secs(30),
115                stream_reconnect_timeout: std::time::Duration::from_secs(30),
116                subscription_stream_restart_delay: Some(std::time::Duration::from_secs(1)),
117                // Allow local clusters to skip the blokli version compatibility check:
118                // some dev images (e.g. bloklid-anvil) do index Safe events but don't
119                // yet advertise the IndexesSafeEvents feature flag in their API response.
120                auto_compatibility_check: std::env::var("HOPR_BLOKLI_NO_COMPAT_CHECK").is_err(),
121                ..Default::default()
122            },
123        ),
124        cfg.hopr.safe_module.module_address,
125    )
126    .await?;
127    chain_connector.connect().await?;
128    let chain_connector = Arc::new(chain_connector);
129
130    let prober_cfg = hopr_ct_full_network::ProberConfig {
131        interval: cfg.hopr.network.probe_interval,
132        shuffle_ttl: cfg.hopr.network.probe_interval * 2,
133        ..Default::default()
134    };
135
136    let node = hopr_reference::build_full_with_chain(
137        &hopr_keys.chain_key,
138        &hopr_keys.packet_key,
139        hopr_lib_cfg,
140        Some(prober_cfg),
141        chain_connector.clone(),
142        HoprServerIpForwardingReactor::new(
143            hopr_keys.packet_key.clone(),
144            cfg.session_ip_forwarding.clone(),
145        ),
146    )
147    .await?;
148
149    if cfg.api.enable {
150        let list = init_rest_api(&cfg, node.clone()).await?;
151        processes.extend_from(list);
152    }
153
154    tracing::debug!("initializing strategies");
155    let mut multi_strategy = crate::strategy::build_strategies(&cfg.strategy, Arc::clone(&node));
156    tracing::debug!(strategy = %multi_strategy, "initialized strategies");
157
158    tracing::debug!("starting up strategies");
159    processes.insert(
160        HoprdProcess::Strategies,
161        tokio::spawn(async move {
162            if let Err(e) = multi_strategy.run().await {
163                tracing::error!(%e, "strategy terminated with error");
164            }
165        }),
166    );
167
168    let mut signals = Signals::new([Signal::Hup, Signal::Int, Signal::Term])
169        .context("failed to register signal handlers")?;
170    while let Some(Ok(signal)) = signals.next().await {
171        match signal {
172            Signal::Hup => {
173                tracing::info!("Received the HUP signal... not doing anything");
174            }
175            Signal::Int | Signal::Term => {
176                tracing::error!(signal = ?signal, "Received a termination signal... tearing down the node");
177                // Explicitly tear down running processes here
178                drop(node);
179                drop(processes);
180
181                tracing::error!(signal = ?signal, "All processes stopped... emulating the default signal handler...");
182                low_level::emulate_default_handler(signal as i32)?;
183                tracing::error!("Shutting down!");
184                break;
185            }
186            _ => {
187                tracing::error!(signal = ?signal, "Received an unhandled signal... emulating the default signal handler...");
188                low_level::emulate_default_handler(signal as i32)?;
189            }
190        }
191    }
192
193    Ok(())
194}

196async fn init_rest_api(
197    cfg: &HoprdConfig,
198    hopr: Arc<HoprNode>,
199) -> anyhow::Result<AbortableList<HoprdProcess>> {
200    let node_cfg_value =
201        serde_json::to_value(cfg.as_redacted()).context("failed to serialize redacted config")?;
202
203    let api_cfg = cfg.api.clone();
204
205    let listen_address = match &cfg.api.host.address {
206        hopr_lib::config::HostType::IPv4(a) | hopr_lib::config::HostType::Domain(a) => {
207            format!("{a}:{}", cfg.api.host.port)
208        }
209    };
210
211    let api_listener = tokio::net::TcpListener::bind(&listen_address)
212        .await
213        .map_err(|e| {
214            hopr_lib::errors::HoprLibError::GeneralError(format!(
215                "REST API bind failed for {listen_address}: {e}"
216            ))
217        })?;
218
219    tracing::info!(listen_address, "Running a REST API");
220
221    let session_listener_sockets = Arc::new(hopr_utils_session::ListenerJoinHandles::default());
222
223    let mut processes = AbortableList::<HoprdProcess>::default();
224
225    processes.insert(
226        HoprdProcess::ListenerSocket,
227        session_listener_sockets.clone(),
228    );
229
230    let cfg_clone = cfg.clone();
231    let (proc, abort_handle) = abortable(
232        async move {
233            if let Err(e) = serve_api(RestApiParameters {
234                listener: api_listener,
235                hoprd_cfg: node_cfg_value,
236                cfg: api_cfg,
237                hopr,
238                session_listener_sockets,
239                default_session_listen_host: cfg_clone
240                    .session_ip_forwarding
241                    .default_entry_listen_host,
242            })
243            .await
244            {
245                tracing::error!(error = %e, "the REST API server could not start")
246            }
247        }
248        .inspect(|_| {
249            tracing::warn!(
250                task = "hoprd - REST API",
251                "long-running background task finished"
252            )
253        }),
254    );
255    let _jh = tokio::spawn(proc);
256    processes.insert(HoprdProcess::RestApi, abort_handle);
257
258    Ok(processes)
259}

261fn update_hopr_lib_config_from_env_vars(cfg: &mut HoprLibConfig) -> anyhow::Result<()> {
262    cfg.protocol.packet.pipeline.output_concurrency = std::env::var("HOPR_INTERNAL_OUT_PACKET_PIPELINE_CONCURRENCY")
263        .ok()
264        .and_then(|p| {
265            p.parse()
266                .inspect_err(
267                    |error| tracing::warn!(%error, "failed to parse HOPR_INTERNAL_OUT_PACKET_PIPELINE_CONCURRENCY"),
268                )
269                .ok()
270        });
271
272    cfg.protocol.packet.pipeline.input_concurrency = std::env::var("HOPR_INTERNAL_IN_PACKET_PIPELINE_CONCURRENCY")
273        .ok()
274        .and_then(|p| {
275            p.parse()
276                .inspect_err(
277                    |error| tracing::warn!(%error, "failed to parse HOPR_INTERNAL_IN_PACKET_PIPELINE_CONCURRENCY"),
278                )
279                .ok()
280        });
281
282    Ok(())
283}

290    static ENV_LOCK: std::sync::Mutex<()> = std::sync::Mutex::new(());
291
292    const OUT_VAR: &str = "HOPR_INTERNAL_OUT_PACKET_PIPELINE_CONCURRENCY";
293    const IN_VAR: &str = "HOPR_INTERNAL_IN_PACKET_PIPELINE_CONCURRENCY";
294
295    struct EnvGuard(Vec<&'static str>);
296
297    impl Drop for EnvGuard {
298        fn drop(&mut self) {
299            for k in &self.0 {
300                // Safety: ENV_LOCK is held for the duration of each test, serializing all env access
301                unsafe { std::env::remove_var(k) };
302            }
303        }
304    }
305
306    fn setup_env(
307        vars: &[(&'static str, Option<&str>)],
308    ) -> (std::sync::MutexGuard<'static, ()>, EnvGuard) {
309        let guard = ENV_LOCK.lock().unwrap_or_else(|e| e.into_inner());
310        for (k, v) in vars {
311            match v {
312                // Safety: ENV_LOCK serializes all env access across tests in this module
313                Some(val) => unsafe { std::env::set_var(k, val) },
314                None => unsafe { std::env::remove_var(k) },
315            }
316        }
317        let keys = vars.iter().map(|(k, _)| *k).collect();
318        (guard, EnvGuard(keys))
319    }
