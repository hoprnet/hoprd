197impl Modify for SecurityAddon {
198    fn modify(&self, openapi: &mut utoipa::openapi::OpenApi) {
199        let components = openapi
200            .components
201            .as_mut()
202            .expect("components should be registered at this point");
203
204        components.add_security_scheme(
205            "bearer_token",
206            SecurityScheme::Http(
207                HttpBuilder::new()
208                    .scheme(HttpAuthScheme::Bearer)
209                    .bearer_format("token")
210                    .description(Some("Bearer token authentication".to_string()))
211                    .build(),
212            ),
213        );
214        components.add_security_scheme(
215            "api_token",
216            SecurityScheme::ApiKey(ApiKey::Header(ApiKeyValue::with_description(
217                "X-Auth-Token",
218                "API Token",
219            ))),
220        );
221    }
222}

234/// Starts the Rest API listener and router.
235pub async fn serve_api<H: HoprNode + hopr_utils_session::SessionFactory>(
236    params: RestApiParameters<H>,
237) -> Result<(), std::io::Error>
238where
239    <<H as hopr_lib::api::node::HasTransportApi>::Transport as hopr_lib::api::node::TransportOperations>::Error:
240        Into<hopr_lib::errors::HoprTransportError>,
241{
242    let RestApiParameters {
243        listener,
244        hoprd_cfg,
245        cfg,
246        hopr,
247        session_listener_sockets,
248        default_session_listen_host,
249    } = params;
250
251    let router = build_api(
252        hoprd_cfg,
253        cfg,
254        hopr,
255        session_listener_sockets,
256        default_session_listen_host,
257    )
258    .await;
259    axum::serve(listener, router).await
260}

262#[allow(clippy::too_many_arguments)]
263async fn build_api<H: HoprNode + hopr_utils_session::SessionFactory>(
264    hoprd_cfg: serde_json::Value,
265    cfg: crate::config::Api,
266    hopr: Arc<H>,
267    open_listeners: Arc<ListenerJoinHandles>,
268    default_listen_host: std::net::SocketAddr,
269) -> Router
270where
271    <<H as hopr_lib::api::node::HasTransportApi>::Transport as hopr_lib::api::node::TransportOperations>::Error:
272        Into<hopr_lib::errors::HoprTransportError>,
273{
274    let state = AppState { hopr };
275    let inner_state = InternalState {
276        auth: Arc::new(cfg.auth.clone()),
277        hoprd_cfg,
278        hopr: state.hopr.clone(),
279        open_listeners,
280        default_listen_host,
281    };
282
283    Router::new()
284        .merge(
285            Router::new()
286                .merge(
287                    SwaggerUi::new("/swagger-ui").url("/api-docs/openapi.json", ApiDoc::openapi()),
288                )
289                .merge(Scalar::with_url("/scalar", ApiDoc::openapi())),
290        )
291        .merge(
292            Router::new()
293                .route("/startedz", get(checks::startedz::<H>))
294                .route("/readyz", get(checks::readyz::<H>))
295                .route("/healthyz", get(checks::healthyz::<H>))
296                .route("/eligiblez", get(checks::eligiblez::<H>))
297                .layer(
298                    ServiceBuilder::new().layer(
299                        CorsLayer::new()
300                            .allow_methods([Method::GET])
301                            .allow_origin(Any)
302                            .allow_headers(Any)
303                            .max_age(std::time::Duration::from_secs(86400)),
304                    ),
305                )
306                .with_state(state.into()),
307        )
308        .merge(
309            Router::new()
310                .route("/metrics", get(root::metrics))
311                .layer(axum::middleware::from_fn_with_state(
312                    inner_state.clone(),
313                    middleware::auth::authenticate::<H>,
314                ))
315                .layer(
316                    ServiceBuilder::new()
317                        .layer(TraceLayer::new_for_http())
318                        .layer(
319                            CorsLayer::new()
320                                .allow_methods([Method::GET])
321                                .allow_origin(Any)
322                                .allow_headers(Any)
323                                .max_age(std::time::Duration::from_secs(86400)),
324                        )
325                        .layer(axum::middleware::from_fn(middleware::prometheus::record))
326                        .layer(CompressionLayer::new())
327                        .layer(ValidateRequestHeaderLayer::accept("text/plain"))
328                        .layer(SetSensitiveRequestHeadersLayer::new([
329                            AUTHORIZATION,
330                            HeaderName::from_static("x-auth-token"),
331                        ])),
332                ),
333        )
334        .nest(
335            BASE_PATH,
336            Router::new()
337                .route("/account/addresses", get(account::addresses::<H>))
338                .route("/account/balances", get(account::balances::<H>))
339                .route("/account/withdraw", post(account::withdraw::<H>))
340                .route("/peers/{address}", get(peers::show_peer_info::<H>))
341                .route("/channels", get(channels::list_channels::<H>))
342                .route("/channels", post(channels::open_channel::<H>))
343                .route("/channels/{address}", get(channels::show_channel::<H>))
344                .route("/channels/{address}", delete(channels::close_channel::<H>))
345                .route(
346                    "/channels/{address}/fund",
347                    post(channels::fund_channel::<H>),
348                )
349                .route("/tickets/redeem", post(tickets::redeem_tickets::<H>))
350                .route(
351                    "/tickets/statistics",
352                    get(tickets::show_ticket_statistics::<H>),
353                )
354                .route("/network/price", get(network::price::<H>))
355                .route("/network/probability", get(network::probability::<H>))
356                .route("/network/connected", get(network::connected::<H>))
357                .route("/network/announced", get(network::announced::<H>))
358                .route("/network/graph", get(network::graph::<H>))
359                .route("/node/version", get(node::version))
360                .route("/node/configuration", get(node::configuration::<H>))
361                .route("/node/info", get(node::info::<H>))
362                .route("/node/status", get(node::status::<H>))
363                .route("/peers/{address}/ping", post(peers::ping_peer::<H>))
364                .route("/session/config/{id}", get(session::session_config::<H>))
365                .route("/session/config/{id}", post(session::adjust_session::<H>))
366                .route("/session/{protocol}", post(session::create_client::<H>))
367                .route("/session/{protocol}", get(session::list_clients::<H>))
368                .route(
369                    "/session/{protocol}/{ip}/{port}",
370                    delete(session::close_client::<H>),
371                )
372                .with_state(inner_state.clone().into())
373                .layer(axum::middleware::from_fn_with_state(
374                    inner_state.clone(),
375                    middleware::auth::authenticate::<H>,
376                ))
377                .layer(
378                    ServiceBuilder::new()
379                        .layer(TraceLayer::new_for_http())
380                        .layer(
381                            CorsLayer::new()
382                                .allow_methods([
383                                    Method::GET,
384                                    Method::POST,
385                                    Method::OPTIONS,
386                                    Method::DELETE,
387                                ])
388                                .allow_origin(Any)
389                                .allow_headers(Any)
390                                .max_age(std::time::Duration::from_secs(86400)),
391                        )
392                        .layer(axum::middleware::from_fn(middleware::prometheus::record))
393                        .layer(CompressionLayer::new())
394                        .layer(ValidateRequestHeaderLayer::accept("application/json"))
395                        .layer(SetSensitiveRequestHeadersLayer::new([
396                            AUTHORIZATION,
397                            HeaderName::from_static("x-auth-token"),
398                        ])),
399                ),
400        )
401}

412/// Standardized error response for the API
413pub(crate) struct ApiError {
414    #[schema(example = "INVALID_INPUT")]
415    pub status: String,
416    #[serde(skip_serializing_if = "Option::is_none")]
417    #[schema(example = "Invalid value passed in parameter 'XYZ'")]
418    pub error: Option<String>,
419}
420
421/// Enumerates all API request errors
422/// Note that `ApiError` should not be instantiated directly, but always rather through the `ApiErrorStatus`.
423#[allow(unused)] // TODO: some errors can no longer be propagated to the REST API
424#[derive(Debug, Clone, PartialEq, Eq, strum::Display)]
425#[strum(serialize_all = "SCREAMING_SNAKE_CASE")]
426enum ApiErrorStatus {
427    InvalidInput,
428    InvalidChannelId,
429    PeerNotFound,
430    ChannelNotFound,
431    TicketsNotFound,
432    Timeout,
433    PingError(String),
434    Unauthorized,
435    InvalidQuality,
436    NotReady,
437    ListenHostAlreadyUsed,
438    SessionNotFound,
439    InvalidSessionId,
440    #[strum(serialize = "UNKNOWN_FAILURE")]
441    UnknownFailure(String),
442}
443
444impl From<ApiErrorStatus> for ApiError {
445    fn from(value: ApiErrorStatus) -> Self {
446        Self {
447            status: value.to_string(),
448            error: if let ApiErrorStatus::UnknownFailure(e) = value {
449                Some(e)
450            } else {
451                None
452            },
453        }
454    }
455}
456
457impl IntoResponse for ApiErrorStatus {
458    fn into_response(self) -> Response {
459        Json(ApiError::from(self)).into_response()
460    }
461}
462
463impl IntoResponse for ApiError {
464    fn into_response(self) -> Response {
465        (StatusCode::INTERNAL_SERVER_ERROR, Json(self)).into_response()
466    }
467}
468
469// Errors lead to `UnknownFailure` per default
470impl<T: Error> From<T> for ApiErrorStatus {
471    fn from(value: T) -> Self {
472        Self::UnknownFailure(value.to_string())
473    }
474}
475
476// Errors lead to `UnknownFailure` per default
477impl<T> From<T> for ApiError
478where
479    T: Error + Into<HoprLibError>,
480{
481    fn from(value: T) -> Self {
482        Self {
483            status: ApiErrorStatus::UnknownFailure("unknown error".to_string()).to_string(),
484            error: Some(value.to_string()),
485        }
486    }
487}
