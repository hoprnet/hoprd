141/// Lists peers with immediate observation data from the network graph.
142///
143/// Returns only peers that have at least one edge with immediate QoS data,
144/// representing nodes the current node has direct transport observations for.
145#[utoipa::path(
146    get,
147    path = const_format::formatcp!("{BASE_PATH}/network/connected"),
148    description = "List connected peers with immediate observation data from the network graph",
149    responses(
150        (status = 200, description = "Connected peers with immediate observations", body = Vec<ConnectedPeerResponse>),
151        (status = 401, description = "Invalid authorization token.", body = ApiError),
152        (status = 422, description = "Unknown failure", body = ApiError)
153    ),
154    security(
155        ("api_token" = []),
156        ("bearer_token" = [])
157    ),
158    tag = "Network"
159)]
160pub(super) async fn connected<
161    H: HasChainApi<ChainError = hopr_lib::errors::HoprLibError> + HasGraphView + Send + Sync + 'static,
162>(
163    State(state): State<Arc<InternalState<H>>>,
164) -> impl IntoResponse {
165    let hopr = state.hopr.clone();
166    let graph = hopr.graph();
167    let edges = graph.connected_edges();
168
169    let me_key = graph.identity();
170
171    // Collect peers that are connected (is_connected == true) with immediate QoS data.
172    let mut peers = Vec::new();
173    for (src, dst, obs) in &edges {
174        if src != me_key {
175            continue;
176        }
177        let Some(imm) = obs.immediate_qos() else {
178            continue;
179        };
180        if !imm.is_connected() {
181            continue;
182        }
183
184        let address = match hopr.chain_api().packet_key_to_chain_key(dst) {
185            Ok(Some(addr)) => addr,
186            _ => continue,
187        };
188
189        peers.push(ConnectedPeerResponse {
190            address,
191            probe_rate: imm.average_probe_rate(),
192            last_update: obs.last_update().as_millis(),
193            average_latency: imm.average_latency().map(|l| l.as_millis()),
194            score: obs.score(),
195        });
196    }
197
198    (StatusCode::OK, Json(peers)).into_response()
199}

271pub(super) async fn announced<
272    H: HasChainApi<ChainError = hopr_lib::errors::HoprLibError> + Send + Sync + 'static,
273>(
274    State(state): State<Arc<InternalState<H>>>,

278    match hopr.announced_peers().await {

281                .into_iter()
282                .map(|entry| AnnouncedPeerResponse {
283                    address: entry.chain_addr,
284                    multiaddrs: entry.get_multiaddrs().to_vec(),
285                    origin: AnnouncementOriginResponse::Chain,
286                })
287                .collect();
288            (StatusCode::OK, Json(response)).into_response()
289        }
290        Err(e) => (
291            StatusCode::UNPROCESSABLE_ENTITY,
292            ApiErrorStatus::UnknownFailure(e.to_string()),
293        )
294            .into_response(),
295    }

304/// Parameters for the network graph endpoint.
305pub(crate) struct GraphQueryRequest {
306    /// When true, only include edges reachable from this node via directed
307    /// traversal. Disconnected subgraphs that cannot be routed through are excluded.
308    #[schema(required = false)]
309    #[serde(default)]
310    reachable_only: bool,
311}
312
313/// Returns the network graph in DOT (Graphviz) format.
314///
315/// Only connected nodes (those with at least one edge) are included.
316/// Nodes are labeled by their on-chain (Ethereum) address when resolvable,
317/// falling back to the offchain public key hex representation.
318/// Edges carry quality annotations: score, latency (ms), and capacity when available.
319///
320/// Pass `?reachableOnly=true` to limit the output to edges reachable from this node.
321#[utoipa::path(
322    get,
323    path = const_format::formatcp!("{BASE_PATH}/network/graph"),
324    description = "Get the network graph in DOT (Graphviz) format",
325    params(GraphQueryRequest),
326    responses(
327        (status = 200, description = "DOT representation of the network graph", body = String, content_type = "text/plain"),
328        (status = 401, description = "Invalid authorization token.", body = ApiError),
329    ),
330    security(
331        ("api_token" = []),
332        ("bearer_token" = [])
333    ),
334    tag = "Network"
335)]
336pub(super) async fn graph<
337    H: HasChainApi<ChainError = hopr_lib::errors::HoprLibError> + HasGraphView + Send + Sync + 'static,
338>(
339    State(state): State<Arc<InternalState<H>>>,
340    Query(query): Query<GraphQueryRequest>,
341) -> impl IntoResponse {
342    let hopr = &state.hopr;
343    let graph = hopr.graph();
344
345    let edges = if query.reachable_only {
346        graph.reachable_edges()
347    } else {
348        graph.connected_edges()
349    };
350
351    // Build offchain key → onchain address mapping for all nodes in the graph.
352    let mut unique_keys = std::collections::HashSet::new();
353    for (src, dst, _) in &edges {
354        unique_keys.insert(*src);
355        unique_keys.insert(*dst);
356    }
357
358    let mut key_to_addr: HashMap<hopr_lib::api::types::crypto::types::OffchainPublicKey, String> =
359        HashMap::new();
360    for key in &unique_keys {
361        let label = match hopr.chain_api().packet_key_to_chain_key(key) {
362            Ok(Some(addr)) => addr.to_string(),
363            _ => key.to_string(),
364        };
365        key_to_addr.insert(*key, label);
366    }
367
368    let label = |key: &hopr_lib::api::types::crypto::types::OffchainPublicKey| {
369        key_to_addr
370            .get(key)
371            .cloned()
372            .unwrap_or_else(|| key.to_string())
373    };
374
375    // Render DOT (Graphviz) format inline using trait methods on the observations.
376    let mut dot = String::from("digraph hopr {\n");
377    for (src, dst, obs) in &edges {
378        let src_label = label(src);
379        let dst_label = label(dst);
380        let mut attrs = vec![format!("score={:.2}", obs.score())];
381        if let Some(imm) = obs.immediate_qos()
382            && let Some(latency) = imm.average_latency()
383        {
384            attrs.push(format!("lat={}ms", latency.as_millis()));
385        }
386        if let Some(inter) = obs.intermediate_qos()
387            && let Some(cap) = inter.capacity()
388        {
389            attrs.push(format!("cap={cap}"));
390        }
391        use std::fmt::Write;
392        let _ = writeln!(
393            dot,
394            "  \"{src_label}\" -> \"{dst_label}\" [label=\"{}\"];",
395            attrs.join(" ")
396        );
397    }
398    dot.push_str("}\n");
399
400    (
401        StatusCode::OK,
402        [(axum::http::header::CONTENT_TYPE, "text/plain")],
403        dot,
404    )
405        .into_response()
406}
