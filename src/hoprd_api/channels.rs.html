156/// the channels in the network as this node sees them.

158        get,
159        path = const_format::formatcp!("{BASE_PATH}/channels"),
160        description = "List channels opened to/from this node. Alternatively, it can print all the channels in the network as this node sees them.",
161        params(ChannelsQueryRequest),
162        responses(
163            (status = 200, description = "Channels fetched successfully", body = NodeChannelsResponse),
164            (status = 401, description = "Invalid authorization token.", body = ApiError),
165            (status = 422, description = "Unknown failure", body = ApiError)

173pub(super) async fn list_channels<
174    H: HasChainApi<ChainError = hopr_lib::errors::HoprLibError> + Send + Sync + 'static,
175>(
176    Query(query): Query<ChannelsQueryRequest>,
177    State(state): State<Arc<InternalState<H>>>,
178) -> impl IntoResponse {
179    let hopr = state.hopr.clone();
180
181    if query.full_topology {
182        let topology = hopr
183            .chain_api()
184            .stream_channels(ChannelSelector::default())
185            .map(|stream| stream.collect::<Vec<_>>());
186
187        match topology {
188            Ok(stream) => {
189                let all = stream
190                    .await
191                    .iter()
192                    .map(channel_to_topology_info)
193                    .collect::<Vec<_>>();
194                (
195                    StatusCode::OK,
196                    Json(NodeChannelsResponse {
197                        incoming: vec![],
198                        outgoing: vec![],
199                        all,
200                    }),
201                )
202                    .into_response()
203            }
204            Err(e) => (
205                StatusCode::UNPROCESSABLE_ENTITY,
206                ApiErrorStatus::from(HoprLibError::chain(e)),
207            )
208                .into_response(),
209        }
210    } else {
211        let me = hopr.identity().node_address;
212        let channels = hopr
213            .channels_to(me)
214            .and_then(|incoming| async {
215                let outgoing = hopr.channels_from(me).await?;
216                Ok((incoming, outgoing))
217            })
218            .await;
219
220        match channels {
221            Ok((incoming, outgoing)) => {
222                let channel_info = NodeChannelsResponse {
223                    incoming: incoming
224                        .into_iter()
225                        .filter(|c| query.including_closed || c.status != ChannelStatus::Closed)
226                        .map(|c| NodeChannel {
227                            id: *c.get_id(),
228                            peer_address: c.source,
229                            status: c.status,
230                            balance: c.balance,
231                        })
232                        .collect(),
233                    outgoing: outgoing
234                        .into_iter()
235                        .filter(|c| query.including_closed || c.status != ChannelStatus::Closed)
236                        .map(|c| NodeChannel {
237                            id: *c.get_id(),
238                            peer_address: c.destination,
239                            status: c.status,
240                            balance: c.balance,
241                        })
242                        .collect(),
243                    all: vec![],
244                };
245
246                (StatusCode::OK, Json(channel_info)).into_response()
247            }
248            Err(e) => (
249                StatusCode::UNPROCESSABLE_ENTITY,
250                ApiErrorStatus::UnknownFailure(e.to_string()),
251            )
252                .into_response(),
253        }
254    }
255}

381/// Filters a channel lookup result to find a non-closed channel.
382///
383/// Returns `ChannelNotFound` when the channel is absent or closed,
384/// and `UnknownFailure` for lookup errors.
385fn filter_open_channel<E: std::error::Error>(
386    result: Result<Option<ChannelEntry>, E>,
387) -> Result<ChannelEntry, ApiErrorStatus> {
388    match result {
389        Ok(Some(ch)) if ch.status != ChannelStatus::Closed => Ok(ch),
390        Ok(_) => Err(ApiErrorStatus::ChannelNotFound),
391        Err(e) => Err(ApiErrorStatus::UnknownFailure(e.to_string())),
392    }
393}

395/// Resolves the channel with the given counterparty in the specified direction.
396fn resolve_channel<H: HasChainApi<ChainError = hopr_lib::errors::HoprLibError> + Send + Sync>(
397    hopr: &H,
398    address: &Address,
399    direction: ChannelDirection,
400) -> Result<ChannelEntry, ApiErrorStatus> {
401    let me = hopr.identity().node_address;
402    let lookup = match direction {
403        ChannelDirection::Outgoing => hopr.channel(me, *address),
404        ChannelDirection::Incoming => hopr.channel(*address, me),
405    };
406    filter_open_channel(lookup)
407}

574        description = "Funds the outgoing channel to the given counterparty with the given amount of HOPR tokens.",
575        params(
576            ("address" = String, Path, description = "On-chain address of the counterparty.", example = "0x188c4462b75e46f0c7262d7f48d182447b93a93c")
577        ),
578        request_body(
579            content = FundBodyRequest,
580            description = "Specifies the amount of HOPR tokens to fund a channel with.",
581            content_type = "application/json",
582        ),
583        responses(
584            (status = 200, description = "Channel funded successfully", body = FundChannelResponse),
585            (status = 400, description = "Invalid counterparty address or request body.", body = ApiError),
586            (status = 401, description = "Invalid authorization token.", body = ApiError),
587            (status = 403, description = "Failed to fund the channel because of insufficient HOPR balance or allowance.", body = ApiError),
588            (status = 404, description = "Channel not found.", body = ApiError),
589            (status = 412, description = "The node is not ready."),
590            (status = 422, description = "Unknown failure", body = ApiError)
591        ),
592        security(
593            ("api_token" = []),
594            ("bearer_token" = [])
595        ),
596        tag = "Channels",
597    )]
598pub(super) async fn fund_channel<
599    H: HasChainApi<ChainError = hopr_lib::errors::HoprLibError> + Send + Sync + 'static,
600>(
601    Path(AddressParams { address }): Path<AddressParams>,
602    State(state): State<Arc<InternalState<H>>>,
603    Json(fund_req): Json<FundBodyRequest>,
604) -> impl IntoResponse {
605    let hopr = state.hopr.clone();
606
607    let channel_id = match resolve_channel(&*hopr, &address, ChannelDirection::Outgoing) {
608        Ok(ch) => *ch.get_id(),
609        Err(status) => {
610            let code = match status {
611                ApiErrorStatus::ChannelNotFound => StatusCode::NOT_FOUND,
612                _ => StatusCode::UNPROCESSABLE_ENTITY,
613            };
614            return (code, status).into_response();
615        }
616    };
617
618    match hopr.fund_channel(&channel_id, fund_req.amount).await {
619        Ok(output) => (
620            StatusCode::OK,
621            Json(FundChannelResponse {
622                hash: *output.tx_hash(),
623            }),
624        )
625            .into_response(),
626        Err(hopr_lib::api::node::EitherErr::Right(HoprLibError::NotReady(..))) => {
627            (StatusCode::PRECONDITION_FAILED, ApiErrorStatus::NotReady).into_response()
628        }
629        Err(e) => (
630            StatusCode::UNPROCESSABLE_ENTITY,
631            ApiErrorStatus::UnknownFailure(e.to_string()),
632        )
633            .into_response(),
634    }
635}
