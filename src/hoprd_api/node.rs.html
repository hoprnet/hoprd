123/// Get information about this HOPR Node.
124#[utoipa::path(
125        get,
126        path = const_format::formatcp!("{BASE_PATH}/node/info"),
127        description = "Get information about this HOPR Node",
128        responses(
129            (status = 200, description = "Fetched node informations", body = NodeInfoResponse),
130            (status = 401, description = "Invalid authorization token.", body = ApiError),
131            (status = 422, description = "Unknown failure", body = ApiError)
132        ),
133        security(
134            ("api_token" = []),
135            ("bearer_token" = [])
136        ),
137        tag = "Node"
138    )]
139pub(super) async fn info<
140    H: HasChainApi<ChainError = hopr_lib::errors::HoprLibError>
141        + HasNetworkView
142        + Send
143        + Sync
144        + 'static,
145>(
146    State(state): State<Arc<InternalState<H>>>,
147) -> Result<impl IntoResponse, ApiError> {
148    let hopr = state.hopr.clone();
149
150    let identity = hopr.identity();
151
152    let provider_url = state
153        .hoprd_cfg
154        .as_object()
155        .and_then(|cfg| cfg.get("blokli_url"))
156        .and_then(|v| v.as_str());
157
158    match futures::try_join!(hopr.chain_info(), hopr.get_channel_closure_notice_period()) {
159        Ok((info, channel_closure_notice_period)) => {
160            let listening: Vec<Multiaddr> =
161                hopr.network_view().listening_as().into_iter().collect();
162            let body = NodeInfoResponse {
163                announced_address: listening.clone(),
164                listening_address: listening,
165                provider_url: provider_url.unwrap_or("n/a").to_owned(),
166                hopr_network_name: info.hopr_network_name,
167                hopr_node_safe: identity.safe_address,
168                connectivity_status: hopr.network_view().health(),
169                chain_status: HasChainApi::status(&*hopr).to_string(),
170                channel_closure_period: channel_closure_notice_period.as_secs(),
171            };
172
173            Ok((StatusCode::OK, Json(body)).into_response())
174        }
175        Err(error) => Ok((
176            StatusCode::UNPROCESSABLE_ENTITY,
177            ApiErrorStatus::from(error),
178        )
179            .into_response()),
180    }
181}

187fn component_status_to_info(status: &ComponentStatus) -> ComponentStatusInfo {
188    match status {
189        ComponentStatus::Ready => ComponentStatusInfo {
190            status: "Ready".into(),
191            detail: None,
192        },
193        ComponentStatus::Initializing(d) => ComponentStatusInfo {
194            status: "Initializing".into(),
195            detail: Some(d.to_string()),
196        },
197        ComponentStatus::Degraded(d) => ComponentStatusInfo {
198            status: "Degraded".into(),
199            detail: Some(d.to_string()),
200        },
201        ComponentStatus::Unavailable(d) => ComponentStatusInfo {
202            status: "Unavailable".into(),
203            detail: Some(d.to_string()),
204        },
205    }
206}

247/// Get the aggregated status of this HOPR node and its individual components.
248#[utoipa::path(
249    get,
250    path = const_format::formatcp!("{BASE_PATH}/node/status"),
251    description = "Get the aggregated status of this HOPR node and its individual components",
252    responses(
253        (status = 200, description = "Fetched node status", body = NodeStatusResponse),
254        (status = 401, description = "Invalid authorization token.", body = ApiError),
255    ),
256    security(
257        ("api_token" = []),
258        ("bearer_token" = [])
259    ),
260    tag = "Node"
261)]
262pub(super) async fn status<
263    H: hopr_lib::api::node::HoprNodeOperations
264        + HasChainApi<ChainError = hopr_lib::errors::HoprLibError>
265        + HasNetworkView
266        + hopr_lib::api::node::HasTransportApi
267        + Send
268        + Sync
269        + 'static,
270>(
271    State(state): State<Arc<InternalState<H>>>,
272) -> impl IntoResponse {
273    use hopr_lib::api::node::{HasTransportApi, HoprNodeOperations};
274
275    let hopr = &state.hopr;
276
277    let chain = HasChainApi::status(&**hopr);
278    let network = HasNetworkView::status(&**hopr);
279    let transport = HasTransportApi::status(&**hopr);
280    let node_state = HoprNodeOperations::status(&**hopr);
281
282    let statuses = [&chain, &network, &transport];
283    let overall = if statuses.iter().any(|s| s.is_unavailable()) {
284        ComponentStatus::Unavailable("one or more components unavailable".into())
285    } else if statuses.iter().any(|s| s.is_degraded()) {
286        ComponentStatus::Degraded("one or more components degraded".into())
287    } else if statuses.iter().any(|s| s.is_initializing()) {
288        ComponentStatus::Initializing("one or more components initializing".into())
289    } else {
290        ComponentStatus::Ready
291    };
292
293    let body = NodeStatusResponse {
294        overall: overall.to_string(),
295        node_state: node_state.to_string(),
296        components: ComponentStatusesResponse {
297            chain: component_status_to_info(&chain),
298            network: component_status_to_info(&network),
299            transport: component_status_to_info(&transport),
300        },
301    };
302
303    (StatusCode::OK, Json(body)).into_response()
304}
